        function showPage(page) {
            document.getElementById('page-atleti').classList.add('hidden');
            document.getElementById('page-gare').classList.add('hidden');
            document.getElementById('page-' + page).classList.remove('hidden');
            
            // Aggiorna icone menu
            const items = document.querySelectorAll('.nav-item');
            items.forEach(i => i.classList.remove('active'));
            event.currentTarget.classList.add('active');
        }

        function filtra(categoria) {
            const stato = document.getElementById('stato-atleti');
            stato.innerText = "Nessun atleta in " + categoria;
            
            const btns = document.querySelectorAll('.tabs button');
            btns.forEach(b => b.classList.remove('active'));
            event.target.classList.add('active');
        }
